
## find max count of jobs that can be performed and max profit

```java
import java.util.ArrayList;

public int[] findMaxProfitAndJobs(int[] ids, int[] deadline, int[] profit) {
    int n = ids.length, maxDeadLine = 0;
    List<Job> jobList = new ArrayList<>();
    Job j;
    for (int i = 0; i < n; i++) {
        j = new Job(ids[i], deadline[i], profit[i]);
        jobList.add(j);
        maxDeadLine = Math.max(maxDeadLine, deadline[i]);
    }
    Arrays.sort(jobList);
    
    boolean[] timeline = new boolean[maxDeadLine + 1];
    int jobsPerformed = 0, maxProfit = 0;
    
    int m;
    for (Job job : jobList) {
        for (int d = job.deadline; d > 0; d--) {
            if (!timeline[d]) {
                timeline[d] = true;
                maxProfit += job.profit;
                jobsPerformed++;
                break;
            }
        }
    }
    
    return new int[] {jobsPerformed, maxProfit};
}

class Job implements Comparable<Job> {
    public int id;
    public int deadline;
    public int profit;

    public Job(int id, int deadline, int profit) {
        this.id = id;
        this.deadline = deadline;
        this.profit = profit;
    }

    @Override
    public int compareTo(Job other) {
        return other.profit - this.profit;
    }
}
```
